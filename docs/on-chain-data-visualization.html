from pathlib import Path
import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.ticker as ticker
import numpy as np

INPUT_DIR  = Path("data/On-chain-measurements-epoch-537-609")
OUTPUT_DIR = Path("data/On-chain-visuals-epoch-537-609")
OUTPUT_DIR.mkdir(parents=True, exist_ok=True)

COLORS = {
    'primary_blue': '#3498db',
    'dark_blue':    '#2c3e50',
    'green':        '#27ae60',
    'red':          '#e74c3c',

def apply_style(ax):
    ax.spines['top'].set_visible(False)
    ax.spines['right'].set_visible(False)
    ax.yaxis.grid(True, alpha=0.3, color='#E0E0E0')
    ax.xaxis.grid(False)

def save_chart(fig, name):
    path = OUTPUT_DIR / f"{name}.png"
    fig.savefig(path, dpi=100, bbox_inches='tight', facecolor='white')
    plt.close(fig)
    print(f"  ✓ Saved {path}")
    return path

def format_large_number(x, pos):
    if x >= 1_000_000: return f'{x/1_000_000:.1f}M'
    if x >= 1_000:     return f'{x/1_000:.0f}K'
    return str(int(x))

def create_active_voting_chart():
    """Multi-line chart showing ADA voting status breakdown over time"""
    df = pd.read_csv(INPUT_DIR / 'ada-owner-metrics/V2 Improved - M3-1-1 Active_voting_pct_537_609.csv')

    # Convert to billions
    for col in ['circulating_ada','active_drep_ada','inactive_drep_ada',
                'no_confidence_ada','abstain_ada','not_delegated_ada']:
        df[col.replace('_ada','_b')] = df[col] / 1e9

    fig, ax1 = plt.subplots(figsize=(14, 8))
    plt.subplots_adjust(bottom=0.15)
    LW = 2.5

    ax1.plot(df['epoch_no'], df['circulating_b'],  color=COLORS['dark_blue'], linewidth=LW, label='Circulating Supply', alpha=0.7)
    ax1.plot(df['epoch_no'], df['not_delegated_b'],color=COLORS['gray'],      linewidth=LW, linestyle='--', label='Not Delegated')
    ax1.plot(df['epoch_no'], df['active_drep_b'],  color=COLORS['green'],     linewidth=LW, label='Active DRep')
    ax1.plot(df['epoch_no'], df['abstain_b'],      color=COLORS['orange'],    linewidth=LW, label='Abstain')
    ax1.plot(df['epoch_no'], df['inactive_drep_b'],color=COLORS['purple'],    linewidth=LW, label='Inactive DRep')
    ax1.plot(df['epoch_no'], df['no_confidence_b'],color=COLORS['red'],       linewidth=LW, label='No Confidence')

    ax1.set_xlabel('Epoch', fontweight='bold')
    ax1.set_ylabel('ADA (Billions)', fontweight='bold', color=COLORS['dark_blue'])
    ax1.yaxis.set_major_formatter(plt.FuncFormatter(lambda x, _: f'{x:.0f}B'))

    ax2 = ax1.twinx()
    ax2.plot(df['epoch_no'], df['active_voting_pct'],
             color=COLORS['teal'], linewidth=LW, marker='o', markersize=4, label='Active Voting %', zorder=10)
    ax2.set_ylabel('Active Voting %\n(Active DRep + No Confidence)',
                   fontweight='bold', color=COLORS['teal'], fontsize=10)
    ax2.set_ylim(0, max(25, df['active_voting_pct'].max() * 1.3))

    ax1.set_title('ADA Voting Status Breakdown by DRep Delegation', fontsize=16, fontweight='bold', pad=20)
    fig.text(0.12, 0.95, 'Total Supply: 45B ADA', ha='left', va='top', fontsize=10, fontweight='bold',
             color=COLORS['dark_blue'],
             bbox=dict(boxstyle='round,pad=0.3', facecolor=COLORS['light_gray'],
                       edgecolor=COLORS['dark_blue'], alpha=0.9, linewidth=1.5))
    apply_style(ax1)

    handles1, labels1 = ax1.get_legend_handles_labels()
    handles2, labels2 = ax2.get_legend_handles_labels()
    ax1.legend(handles1+handles2, labels1+labels2,
               loc='upper center', bbox_to_anchor=(0.5,-0.08), ncol=4, frameon=True)

    return save_chart(fig, 'ada-owner-metrics/V2 Improved - M3-1-1 Active_voting_pct')

def create_unique_delegators_chart():
    """Cumulative growth chart of unique stake addresses delegated to DReps"""
    df = pd.read_csv(INPUT_DIR / 'ada-owner-metrics/M3-1-2 DRep_unique_delegators_537_609.csv')

    fig, ax = plt.subplots(figsize=(12, 6))
    ax.fill_between(df['epoch_no'], 0, df['total_unique_stake_addresses'],
                    alpha=0.3, color=COLORS['green'])
    ax.plot(df['epoch_no'], df['total_unique_stake_addresses'],
            color=COLORS['green'], linewidth=2.5)

    ax.set_xlabel('Epoch', fontweight='bold')
    ax.set_ylabel('Unique Stake Addresses', fontweight='bold')
    ax.set_title('Unique Stake Addresses Delegated to DReps', fontsize=16, fontweight='bold', pad=20)
    ax.yaxis.set_major_formatter(ticker.FuncFormatter(format_large_number))
    ax.set_ylim(0, df['total_unique_stake_addresses'].max() * 1.1)
    ax.set_xlim(df['epoch_no'].min(), df['epoch_no'].max())

    start_val = df.iloc[0]['total_unique_stake_addresses']
    end_val   = df.iloc[-1]['total_unique_stake_addresses']
    growth_pct = ((end_val - start_val) / start_val) * 100
    ax.annotate(f'{end_val:,.0f}\n(+{growth_pct:.0f}% growth)',
                xy=(df.iloc[-1]['epoch_no'], end_val),
                xytext=(-60, 10), textcoords='offset points',
                fontsize=10, fontweight='bold', color=COLORS['green'],
                bbox=dict(boxstyle='round,pad=0.3', facecolor='white', edgecolor=COLORS['green'], alpha=0.8),
                arrowprops=dict(arrowstyle='->', color=COLORS['green']))

    apply_style(ax)
    plt.tight_layout()
    return save_chart(fig, 'ada-owner-metrics/M3-1-2 unique_delegators')

def create_drep_stickiness_chart():
    """DRep delegation churn rate with moving average"""
    df = pd.read_csv(INPUT_DIR / 'ada-owner-metrics/M3-1-3 DRep_stickiness_537_609.csv')
    df['epoch'] = df['epoch_window'].str.split(' -> ').str[0].astype(int)

    fig, ax = plt.subplots(figsize=(12, 6))
    ax.plot(df['epoch'], df['churn_pct'],
            color=COLORS['primary_blue'], linewidth=1, alpha=0.5, label='Raw Data')

    window = 5
    df['churn_ma'] = df['churn_pct'].rolling(window=window, center=True).mean()
    ax.plot(df['epoch'], df['churn_ma'],
            color=COLORS['primary_blue'], linewidth=2.5, label=f'{window}-Epoch Moving Avg')

    avg_churn = df['churn_pct'].mean()
    ax.axhline(y=avg_churn, color=COLORS['orange'], linestyle=':',
               linewidth=1.5, alpha=0.7, label=f'Average ({avg_churn:.2f}%)')

    ax.set_xlabel('Epoch', fontweight='bold')
    ax.set_ylabel('Addresses Changing Delegation (%)', fontweight='bold')
    ax.set_title('DRep Delegation Change Rate (Stickiness)', fontsize=16, fontweight='bold', pad=20)
    ax.set_ylim(0, df['churn_pct'].max() * 1.2)
    ax.legend(loc='upper right', frameon=True)
    apply_style(ax)
    plt.tight_layout()
    return save_chart(fig, 'ada-owner-metrics/M3-1-3 drep_stickiness')

def create_gini_chart():
    """Gini coefficient time series with real-world and DAO comparisons"""
    df = pd.read_csv(INPUT_DIR / 'DRep-Metrics/M3-2-1 DRep_gini_with_noconf_537_609.csv')

    fig, ax = plt.subplots(figsize=(14, 8))

    REAL_WORLD_HIGHEST = 0.63   # South Africa
    DAO_AVERAGE = 0.97; DAO_AAVE = 0.98; DAO_UNISWAP = 0.99; DAO_COMPOUND = 0.99

    # Shaded zones
    ax.axhspan(0.50, REAL_WORLD_HIGHEST, alpha=0.08, color=COLORS['green'],  label='Real-World Range')
    ax.axhspan(REAL_WORLD_HIGHEST, DAO_AVERAGE, alpha=0.06, color=COLORS['amber'])
    ax.axhspan(DAO_AVERAGE, 1.0,  alpha=0.08, color=COLORS['red'],   label='DAO Typical Range')

    # Reference lines
    ax.axhline(REAL_WORLD_HIGHEST, color=COLORS['green'],  linestyle='--', linewidth=1.5, alpha=0.8, label='South Africa (0.63)')
    ax.axhline(DAO_AVERAGE,        color=COLORS['red'],    linestyle='--', linewidth=1.5, alpha=0.8, label='DAO Avg (0.97)')
    ax.axhline(DAO_AAVE,           color=COLORS['purple'], linestyle=':',  linewidth=1.5, alpha=0.7, label='Aave (0.98)')
    ax.axhline(DAO_UNISWAP,        color=COLORS['orange'], linestyle=':',  linewidth=1.5, alpha=0.7, label='Uniswap (0.99)')
    ax.axhline(DAO_COMPOUND,       color=COLORS['teal'],   linestyle=':',  linewidth=1.5, alpha=0.7, label='Compound (0.99)')

    # Cardano DRep line
    ax.plot(df['epoch_no'], df['gini_coefficient'],
            color=COLORS['primary_blue'], linewidth=2.5, marker='o', markersize=4, label='Cardano DRep Gini', zorder=10)

    ax.set_ylim(0.5, 1.0)
    ax.set_xlim(df['epoch_no'].min(), df['epoch_no'].max())
    ax.set_xlabel('Epoch', fontweight='bold')
    ax.set_ylabel('Gini Coefficient', fontweight='bold')
    ax.set_title('Gini Coefficient of DRep Voting Power Distribution', fontsize=16, fontweight='bold', pad=20)

    # Zone labels on right edge
    ax.text(df['epoch_no'].max() + 0.5, 0.56, 'Real-World\nRange', fontsize=9, color=COLORS['green'], va='center')
    ax.text(df['epoch_no'].max() + 0.5, 0.985,'DAO Average',       fontsize=9, color=COLORS['red'],   va='center')

    ax.legend(loc='lower left', frameon=True, fontsize=9)
    apply_style(ax)
    plt.tight_layout()
    return save_chart(fig, 'DRep-metrics/V2 Improved - M3-2-1_gini_coefficient')

def create_nakamoto_chart():
    """Nakamoto coefficient bar chart showing DRep decentralisation over time"""
    df = pd.read_csv(INPUT_DIR / 'DRep-Metrics/M3-2-2 DRep_nakamoto_537_609.csv')
    # Keep one row per epoch (the Nakamoto value for that epoch)
    df_epoch = df.drop_duplicates('epoch_no').sort_values('epoch_no')

    fig, ax = plt.subplots(figsize=(14, 6))
    ax.bar(df_epoch['epoch_no'], df_epoch['nakamoto_coefficient'],
           color=COLORS['primary_blue'], alpha=0.7, width=0.8, label='Nakamoto Coefficient')
    ax.step(df_epoch['epoch_no'], df_epoch['nakamoto_coefficient'],
            color=COLORS['dark_blue'], linewidth=2, where='mid')

    ax.set_xlabel('Epoch', fontweight='bold')
    ax.set_ylabel('Nakamoto Coefficient (# DReps needed for >50% vote)', fontweight='bold')
    ax.set_title('Nakamoto Coefficient – DRep Voting Power', fontsize=16, fontweight='bold', pad=20)
    ax.yaxis.set_major_locator(ticker.MaxNLocator(integer=True))

    # Annotate peak
    peak_row = df_epoch.loc[df_epoch['nakamoto_coefficient'].idxmax()]
    ax.annotate(f"Peak: {int(peak_row['nakamoto_coefficient'])}",
                xy=(peak_row['epoch_no'], peak_row['nakamoto_coefficient']),
                xytext=(8, 6), textcoords='offset points',
                fontsize=9, color=COLORS['dark_blue'], fontweight='bold')

    apply_style(ax)
    plt.tight_layout()
    return save_chart(fig, 'DRep-metrics/M3-2-2 Nakamoto_Coefficient')

def create_cc_abstain_chart():
    """Stacked bar (Yes/No/Abstain) with abstain rate % overlay per epoch"""
    df = pd.read_csv(INPUT_DIR / 'CC-Metrics/M3-4-3 CC_abstain_rate_537_609.csv')
    df = df[df['category'].str.startswith('Epoch:')].copy()
    df = df.sort_values('epoch_no')

    fig, ax1 = plt.subplots(figsize=(14, 7))
    x = np.arange(len(df))

    ax1.bar(x, df['yes_count'],     label='Yes (Constitutional)',   color=COLORS['green'], alpha=0.85)
    ax1.bar(x, df['no_count'],      bottom=df['yes_count'],         label='No (Unconstitutional)', color=COLORS['red'],    alpha=0.85)
    ax1.bar(x, df['abstain_count'], bottom=df['yes_count']+df['no_count'], label='Abstain', color=COLORS['orange'], alpha=0.85)

    ax1.set_xticks(x)
    ax1.set_xticklabels(df['epoch_no'].astype(int), rotation=45, ha='right')
    ax1.set_xlabel('Epoch', fontweight='bold')
    ax1.set_ylabel('Number of CC Votes', fontweight='bold')

    ax2 = ax1.twinx()
    ax2.plot(x, df['abstain_rate_pct'], color=COLORS['purple'],
             linewidth=2.5, marker='o', markersize=5, label='Abstain Rate %', zorder=10)
    ax2.set_ylabel('Abstain Rate (%)', fontweight='bold', color=COLORS['purple'])
    ax2.set_ylim(0, df['abstain_rate_pct'].max() * 2)

    ax1.set_title('Constitutional Committee Vote Breakdown & Abstain Rate',
                  fontsize=16, fontweight='bold', pad=20)
    h1, l1 = ax1.get_legend_handles_labels()
    h2, l2 = ax2.get_legend_handles_labels()
    ax1.legend(h1+h2, l1+l2, loc='upper left', frameon=True)
    apply_style(ax1)
    plt.tight_layout()
    return save_chart(fig, 'CC-metrics/M3-4-3 cc_abstain_rate')
