Systemd Service

Create and enable the cardano-node systemd service.

bash 
sudo tee /etc/systemd/system/cardano-node.service > /dev/null << 'EOF'
[Unit]
Description=Cardano Node (Mainnet)
After=network-online.target
Wants=network-online.target

[Service]
Type=simple
User=cardano
Group=cardano
WorkingDirectory=/var/lib/cardano-node
UMask=0007
ExecStart=/usr/local/bin/cardano-node run \
  --config /etc/cardano/config.json \
  --topology /etc/cardano/topology.json \
  --database-path /var/lib/cardano-node/db \
  --socket-path /run/cardano-node/node.socket \
  --host-addr 0.0.0.0 \
  --port 3001
Restart=always
RestartSec=10
LimitNOFILE=1048576
Nice=-5
NoNewPrivileges=true
PrivateTmp=true
ProtectSystem=full
ProtectHome=true
ReadWritePaths=/var/lib/cardano-node /run/cardano-node /var/log/cardano-node

[Install]
WantedBy=multi-user.target
EOF

sudo systemctl daemon-reload
sudo systemctl enable cardano-node

Governance Metrics Queries

SQL queries for calculating governance metrics from the cardano-db-sync database, aligned with the Cardano Governance Measurement Framework. All results export to CSV files in /tmp/cardano_exports/.

4.1.1 — Percentage of Circulating ADA Actively Voting (Excluding Abstain) → M3-1-1_Active_voting_pct_537_609.csv
SQL 
COPY (
WITH
  epoch_supply AS (
    SELECT epoch_no,
      utxo / 1000000.0 AS circulating_ada,
      reserves / 1000000.0 AS reserves_ada
    FROM ada_pots
    WHERE epoch_no BETWEEN 537 AND 609
  ),
  drep_breakdown AS (
    SELECT dd.epoch_no,
      SUM(CASE WHEN dh.view = 'drep_always_abstain' THEN dd.amount ELSE 0 END) / 1000000.0 AS abstain_ada,
      SUM(CASE WHEN dh.view = 'drep_always_no_confidence' THEN dd.amount ELSE 0 END) / 1000000.0 AS no_confidence_ada,
      SUM(CASE WHEN dh.view NOT IN ('drep_always_abstain','drep_always_no_confidence')
               AND dd.active_until >= dd.epoch_no THEN dd.amount ELSE 0 END) / 1000000.0 AS active_drep_ada,
      SUM(CASE WHEN dh.view NOT IN ('drep_always_abstain','drep_always_no_confidence')
               AND dd.active_until < dd.epoch_no THEN dd.amount ELSE 0 END) / 1000000.0 AS inactive_drep_ada
    FROM drep_distr dd
    JOIN drep_hash dh ON dd.hash_id = dh.id
    WHERE dd.epoch_no BETWEEN 537 AND 609
    GROUP BY dd.epoch_no
  )
SELECT
  es.epoch_no,
  ROUND(es.circulating_ada, 2) AS circulating_ada,
  ROUND(COALESCE(db.abstain_ada, 0), 2) AS abstain_ada,
  ROUND(COALESCE(db.no_confidence_ada, 0), 2) AS no_confidence_ada,
  ROUND(COALESCE(db.active_drep_ada, 0), 2) AS active_drep_ada,
  ROUND(COALESCE(db.inactive_drep_ada, 0), 2) AS inactive_drep_ada,
  ROUND(es.circulating_ada - COALESCE(db.abstain_ada,0)
    - COALESCE(db.no_confidence_ada,0)
    - COALESCE(db.active_drep_ada,0)
    - COALESCE(db.inactive_drep_ada,0), 2) AS not_delegated_ada,
  ROUND(((COALESCE(db.active_drep_ada,0) + COALESCE(db.no_confidence_ada,0))
    / es.circulating_ada) * 100, 2) AS active_voting_pct
FROM epoch_supply es
LEFT JOIN drep_breakdown db ON es.epoch_no = db.epoch_no
ORDER BY es.epoch_no ASC
) TO '/tmp/cardano_exports/M3-1-1 Active_voting_pct_537_609.csv' WITH CSV HEADER;
4.1.2 — Number of Unique Stake Addresses Delegated to DReps → M3-1-2_DRep_unique_delegators_537_609.csv
SQL 
COPY (
WITH
  all_delegations AS (
    SELECT dv.addr_id, dv.drep_hash_id, dh.view AS drep_view,
      b.epoch_no AS delegation_epoch, b.slot_no, tx.block_index
    FROM delegation_vote dv
    JOIN drep_hash dh ON dv.drep_hash_id = dh.id
    JOIN tx ON dv.tx_id = tx.id
    JOIN block b ON tx.block_id = b.id
    WHERE b.epoch_no <= 609
  ),
  drep_first_registration AS (
    SELECT dr.drep_hash_id, MIN(b.epoch_no) AS first_registered_epoch
    FROM drep_registration dr
    JOIN tx ON dr.tx_id = tx.id
    JOIN block b ON tx.block_id = b.id
    WHERE dr.deposit > 0
    GROUP BY dr.drep_hash_id
  ),
  epochs AS (SELECT generate_series(537, 609) AS epoch_no),
  epoch_delegations AS (
    SELECT e.epoch_no, ad.addr_id, ad.drep_hash_id, ad.drep_view,
      ROW_NUMBER() OVER (PARTITION BY e.epoch_no, ad.addr_id
        ORDER BY ad.slot_no DESC, ad.block_index DESC) AS rn
    FROM epochs e
    JOIN all_delegations ad ON ad.delegation_epoch <= e.epoch_no
  ),
  latest_per_epoch AS (
    SELECT epoch_no, addr_id, drep_hash_id, drep_view
    FROM epoch_delegations WHERE rn = 1
  ),
  final_counts AS (
    SELECT lpe.epoch_no, COUNT(DISTINCT lpe.addr_id) AS unique_delegators
    FROM latest_per_epoch lpe
    LEFT JOIN drep_first_registration dfr ON lpe.drep_hash_id = dfr.drep_hash_id
    WHERE lpe.drep_view IS NOT NULL
      AND lpe.drep_view != 'drep_always_abstain'
      AND (lpe.drep_view = 'drep_always_no_confidence'
           OR dfr.first_registered_epoch <= lpe.epoch_no)
    GROUP BY lpe.epoch_no
  )
SELECT epoch_no, unique_delegators AS total_unique_stake_addresses
FROM final_counts ORDER BY epoch_no ASC
) TO '/tmp/cardano_exports/M3-1-2 DRep_unique_delegators_537_609.csv' WITH CSV HEADER;
4.1.3 — Stickiness of ADA Owner to DRep Delegation → M3-1-3_DRep_stickiness_537_609.csv
SQL 
COPY (
WITH
  epochs AS (SELECT generate_series(537, 608) AS epoch_no),
  delegation_snapshots AS (
    SELECT e.epoch_no, dv.addr_id, dv.drep_hash_id, dv.drep_view
    FROM epochs e
    CROSS JOIN LATERAL (
      SELECT DISTINCT ON (dv_inner.addr_id)
        dv_inner.addr_id, dv_inner.drep_hash_id, dh.view AS drep_view
      FROM delegation_vote dv_inner
      JOIN drep_hash dh ON dv_inner.drep_hash_id = dh.id
      JOIN tx ON dv_inner.tx_id = tx.id
      JOIN block b ON tx.block_id = b.id
      WHERE b.epoch_no <= e.epoch_no
      ORDER BY dv_inner.addr_id, b.slot_no DESC, tx.block_index DESC
    ) dv
    WHERE dv.drep_view != 'drep_always_abstain'
  ),
  stickiness_calc AS (
    SELECT s1.epoch_no AS from_epoch, s1.epoch_no + 1 AS to_epoch,
      COUNT(DISTINCT s1.addr_id) AS total_delegated_at_e0,
      COUNT(DISTINCT CASE WHEN s2.addr_id IS NOT NULL THEN s1.addr_id END) AS still_delegated,
      COUNT(DISTINCT CASE WHEN s2.addr_id IS NOT NULL AND s1.drep_hash_id != s2.drep_hash_id THEN s1.addr_id END) AS changed_drep,
      COUNT(DISTINCT CASE WHEN s2.addr_id IS NULL THEN s1.addr_id END) AS left_delegation
    FROM delegation_snapshots s1
    LEFT JOIN delegation_snapshots s2
      ON s1.addr_id = s2.addr_id AND s2.epoch_no = s1.epoch_no + 1
    WHERE s1.epoch_no <= 608
    GROUP BY s1.epoch_no
  )
SELECT
  from_epoch || ' -> ' || to_epoch AS epoch_window,
  total_delegated_at_e0, still_delegated, changed_drep, left_delegation,
  CASE WHEN total_delegated_at_e0 > 0
    THEN ROUND((changed_drep::numeric / total_delegated_at_e0::numeric) * 100, 2)
    ELSE 0 END AS churn_pct,
  CASE WHEN still_delegated > 0
    THEN ROUND((changed_drep::numeric / still_delegated::numeric) * 100, 2)
    ELSE 0 END AS churn_pct_among_active
FROM stickiness_calc ORDER BY from_epoch ASC
) TO '/tmp/cardano_exports/M3-1-3 DRep_stickiness_537_609.csv' WITH CSV HEADER;
4.2.1 — Gini Coefficient of Delegated Voting Power Among DReps → M3-2-1_DRep_gini_537_609.csv
SQL 
COPY (
WITH
  drep_stake_list AS (
    SELECT dd.epoch_no, dd.amount AS stake_lovelace
    FROM drep_distr dd
    JOIN drep_hash dh ON dd.hash_id = dh.id
    WHERE dd.epoch_no BETWEEN 537 AND 609
      AND dh.view != 'drep_always_abstain'
      AND dd.amount > 0
  ),
  gini_math AS (
    SELECT epoch_no, stake_lovelace,
      ROW_NUMBER() OVER (PARTITION BY epoch_no ORDER BY stake_lovelace ASC) AS rank,
      COUNT(*) OVER (PARTITION BY epoch_no) AS n_count,
      SUM(stake_lovelace) OVER (PARTITION BY epoch_no) AS total_stake
    FROM drep_stake_list
  )
SELECT epoch_no,
  ROUND(
    CASE WHEN SUM(stake_lovelace) > 0
      THEN SUM((2.0 * rank - n_count - 1) * stake_lovelace) / (n_count * SUM(stake_lovelace)::numeric)
      ELSE 0 END, 4) AS gini_coefficient,
  n_count AS active_voting_options_count,
  ROUND(SUM(stake_lovelace) / 1000000.0, 2) AS total_stake_ada
FROM gini_math
GROUP BY epoch_no, n_count
ORDER BY epoch_no ASC
) TO '/tmp/cardano_exports/M3-2-1 DRep_gini_with_noconf_537_609.csv' WITH CSV HEADER;
4.2.2 — Nakamoto Coefficient for DRep Power → M3-2-2_DRep_nakamoto_537_609.csv
SQL 
COPY (
WITH
  governance_thresholds AS (
    SELECT 0.51 AS threshold, 'Info Action' AS actions
    UNION ALL SELECT 0.60, 'Hard Fork Initiation; New Committee (No Confidence State)'
    UNION ALL SELECT 0.67, 'Motion of No Confidence; New Committee (Normal); Treasury Withdrawal; Network/Economic/Technical Parameter Changes'
    UNION ALL SELECT 0.75, 'Update Constitution; Governance Parameter Changes'
  ),
  drep_stake_list AS (
    SELECT dd.epoch_no, dh.id AS drep_id, dh.view AS drep_view, dd.amount AS stake_lovelace
    FROM drep_distr dd
    JOIN drep_hash dh ON dd.hash_id = dh.id
    WHERE dd.epoch_no BETWEEN 537 AND 609
      AND dh.view != 'drep_always_abstain'
      AND dd.amount > 0
  ),
  total_active_stake AS (
    SELECT epoch_no, SUM(stake_lovelace) AS total_stake, COUNT(DISTINCT drep_id) AS total_dreps
    FROM drep_stake_list GROUP BY epoch_no
  ),
  ranked_dreps AS (
    SELECT dsl.epoch_no, dsl.drep_id, dsl.drep_view, dsl.stake_lovelace,
      tas.total_stake, tas.total_dreps,
      SUM(dsl.stake_lovelace) OVER (PARTITION BY dsl.epoch_no
        ORDER BY dsl.stake_lovelace DESC, dsl.drep_id ASC
        ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW) AS cumulative_stake,
      ROW_NUMBER() OVER (PARTITION BY dsl.epoch_no
        ORDER BY dsl.stake_lovelace DESC, dsl.drep_id ASC) AS drep_rank
    FROM drep_stake_list dsl JOIN total_active_stake tas ON dsl.epoch_no = tas.epoch_no
  ),
  nakamoto_per_threshold AS (
    SELECT rd.epoch_no, gt.threshold, gt.actions, rd.total_stake, rd.total_dreps,
      MIN(rd.drep_rank) AS nakamoto_coefficient
    FROM ranked_dreps rd CROSS JOIN governance_thresholds gt
    WHERE rd.cumulative_stake >= (rd.total_stake * gt.threshold)

4.2.4 — Net Change in Active DRep Count → M3-2-4_DRep_net_change_537_609.csv
SQL 
COPY (
WITH
  measurement_windows AS (
    SELECT window_start, window_start + 11 AS window_end,
      window_start || '-' || (window_start + 11) AS epoch_window
    FROM generate_series(537, 597, 12) AS window_start
  ),
  registration_events AS (
    SELECT dr.drep_hash_id, dh.view AS drep_view, b.epoch_no, dr.deposit,
      CASE WHEN dr.deposit > 0 THEN 'registration'
           WHEN dr.deposit < 0 THEN 'retirement' ELSE 'update' END AS event_type,
      ROW_NUMBER() OVER (PARTITION BY dr.drep_hash_id ORDER BY b.epoch_no, tx.block_index) AS reg_sequence
    FROM drep_registration dr
    JOIN drep_hash dh ON dr.drep_hash_id = dh.id
    JOIN tx ON dr.tx_id = tx.id JOIN block b ON tx.block_id = b.id
    WHERE dh.view NOT IN ('drep_always_abstain', 'drep_always_no_confidence')
  ),
  new_registrations AS (
    SELECT mw.epoch_window, mw.window_start, COUNT(DISTINCT re.drep_hash_id) AS new_dreps
    FROM measurement_windows mw
    JOIN registration_events re ON re.epoch_no BETWEEN mw.window_start AND mw.window_end
      AND re.reg_sequence = 1 AND re.event_type = 'registration'
    GROUP BY mw.epoch_window, mw.window_start
  ),
  voluntary_retirements AS (
    SELECT mw.epoch_window, mw.window_start, COUNT(*) AS retired_dreps
    FROM measurement_windows mw
    JOIN registration_events re ON re.epoch_no BETWEEN mw.window_start AND mw.window_end
      AND re.event_type = 'retirement'
    GROUP BY mw.epoch_window, mw.window_start
  ),
  combined_metrics AS (
    SELECT mw.epoch_window, mw.window_start, mw.window_end,
      COALESCE(nr.new_dreps, 0) AS registrations,
      COALESCE(vr.retired_dreps, 0) AS voluntary_retirements
    FROM measurement_windows mw
    LEFT JOIN new_registrations nr ON mw.window_start = nr.window_start
    LEFT JOIN voluntary_retirements vr ON mw.window_start = vr.window_start
  )
SELECT epoch_window, window_start, window_end,
  registrations AS new_registrations_R,
  voluntary_retirements AS retirements_D,
  registrations - voluntary_retirements AS net_change,
  ROUND((registrations - voluntary_retirements)::numeric / 12.0, 4) AS net_change_rate_per_epoch
FROM combined_metrics ORDER BY window_start ASC
) TO '/tmp/cardano_exports/M3-2-4 DRep_net_change_537_609.csv' WITH CSV HEADER;

4.3.2 — Stickiness of ADA Delegated to SPOs → M3-3-2_SPO_stickiness_537_609.csv
SQL 
COPY (
WITH
  measurement_intervals AS (SELECT 3 AS interval_epochs UNION ALL SELECT 6 UNION ALL SELECT 12),
  epoch_pairs AS (
    SELECT e0.epoch_no AS epoch_e0, e0.epoch_no + mi.interval_epochs AS epoch_e1,
      mi.interval_epochs AS interval_length
    FROM generate_series(537, 609) AS e0(epoch_no)
    CROSS JOIN measurement_intervals mi
    WHERE e0.epoch_no + mi.interval_epochs <= 609
  ),
  stake_at_e0 AS (
    SELECT ep.epoch_e0, ep.epoch_e1, ep.interval_length,
      es.addr_id, es.pool_id AS pool_id_at_e0, es.amount AS stake_at_e0
    FROM epoch_pairs ep JOIN epoch_stake es ON es.epoch_no = ep.epoch_e0
    WHERE es.amount > 0
  ),
  stake_comparison AS (
    SELECT s0.*, es1.pool_id AS pool_id_at_e1, es1.amount AS stake_at_e1
    FROM stake_at_e0 s0
    LEFT JOIN epoch_stake es1 ON s0.addr_id = es1.addr_id AND es1.epoch_no = s0.epoch_e1
  ),
  stake_classification AS (
    SELECT *,
      CASE WHEN pool_id_at_e0 = pool_id_at_e1 THEN stake_at_e0 ELSE 0 END AS sticky_stake,
      CASE WHEN pool_id_at_e0 != pool_id_at_e1 OR pool_id_at_e1 IS NULL THEN stake_at_e0 ELSE 0 END AS moved_stake
    FROM stake_comparison
  ),
  stickiness_stats AS (
    SELECT epoch_e0, epoch_e1, interval_length,
      COUNT(DISTINCT addr_id) AS total_delegators_at_e0,
      SUM(stake_at_e0) AS total_stake_at_e0,
      COUNT(DISTINCT CASE WHEN sticky_stake > 0 THEN addr_id END) AS sticky_delegators,
      SUM(sticky_stake) AS total_sticky_stake,
      SUM(moved_stake) AS total_moved_stake
    FROM stake_classification GROUP BY epoch_e0, epoch_e1, interval_length
  )
SELECT epoch_e0 || ' -> ' || epoch_e1 AS epoch_window,
  interval_length || ' epochs' AS measurement_interval,
  total_delegators_at_e0,
  ROUND(total_stake_at_e0 / 1000000.0, 2) AS total_stake_at_e0_ada,
  ROUND(total_sticky_stake / 1000000.0, 2) AS sticky_stake_ada,
  CASE WHEN total_stake_at_e0 > 0
    THEN ROUND((total_sticky_stake::numeric / total_stake_at_e0::numeric) * 100, 2) ELSE 0
  END AS sticky_stake_pct,
  ROUND(total_moved_stake / 1000000.0, 2) AS moved_stake_ada,
  CASE WHEN total_stake_at_e0 > 0
    THEN ROUND((total_moved_stake::numeric / total_stake_at_e0::numeric) * 100, 2) ELSE 0
  END AS moved_stake_pct
FROM stickiness_stats ORDER BY interval_length ASC, epoch_e0 ASC
) TO '/tmp/cardano_exports/M3-3-2 SPO_stickiness_537_609.csv' WITH CSV HEADER;
4.4.1 — Average CC Response Time to Governance Actions → M3-4-1_CC_response_time_537_609.csv
SQL 
COPY (
WITH
  eligible_governance_actions AS (
    SELECT ga.id AS gov_action_id, ga.type AS action_type,
      b.slot_no AS submission_slot, b.epoch_no AS submission_epoch
    FROM gov_action_proposal ga
    JOIN tx ON ga.tx_id = tx.id JOIN block b ON tx.block_id = b.id
    WHERE b.epoch_no BETWEEN 537 AND 609
      AND ga.type NOT IN ('NoConfidence', 'NewCommittee')
  ),
  cc_votes AS (
    SELECT ega.gov_action_id, ega.action_type, ega.submission_slot, ega.submission_epoch,
      vp.committee_voter AS cc_member_id,
      b_vote.slot_no AS vote_slot, b_vote.epoch_no AS vote_epoch, vp.vote AS vote_cast
    FROM eligible_governance_actions ega
    JOIN voting_procedure vp ON vp.gov_action_proposal_id = ega.gov_action_id
    JOIN tx tx_vote ON vp.tx_id = tx_vote.id JOIN block b_vote ON tx_vote.block_id = b_vote.id
    WHERE vp.committee_voter IS NOT NULL
  ),
  action_vote_stats AS (
    SELECT ega.gov_action_id, ega.action_type, ega.submission_slot, ega.submission_epoch,
      COUNT(DISTINCT ccv.cc_member_id) AS cc_members_voted,
      MIN(ccv.vote_slot) AS first_vote_slot, MAX(ccv.vote_slot) AS last_vote_slot
    FROM eligible_governance_actions ega
    LEFT JOIN cc_votes ccv ON ega.gov_action_id = ccv.gov_action_id
    GROUP BY ega.gov_action_id, ega.action_type, ega.submission_slot, ega.submission_epoch
  ),
  action_response_stats AS (
    SELECT avs.*, 7 AS expected_cc_members,
      CASE WHEN avs.cc_members_voted > 0 THEN avs.first_vote_slot - avs.submission_slot ELSE 2592000 END AS first_response_slots,
      CASE WHEN avs.cc_members_voted > 0 THEN avs.last_vote_slot - avs.submission_slot ELSE 2592000 END AS last_response_slots,
      (avs.cc_members_voted::numeric / 7::numeric * 100) AS participation_pct
    FROM action_vote_stats avs
  ),
  overall_stats AS (
    SELECT 'ALL_ACTIONS'::text AS action_category, COUNT(*) AS action_count,
      SUM(cc_members_voted) AS total_votes_cast,
      ROUND(AVG(cc_members_voted), 2) AS avg_cc_votes_per_action,
      ROUND(AVG(participation_pct), 2) AS avg_cc_participation_pct,
      PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY first_response_slots) AS median_first_response_slots,
      PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY last_response_slots) AS median_last_response_slots
    FROM action_response_stats
  ),
  by_action_type AS (
    SELECT action_type::text AS action_category, COUNT(*) AS action_count,
      SUM(cc_members_voted) AS total_votes_cast,
      ROUND(AVG(cc_members_voted), 2) AS avg_cc_votes_per_action,
      ROUND(AVG(participation_pct), 2) AS avg_cc_participation_pct,
      PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY first_response_slots) AS median_first_response_slots,
      PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY last_response_slots) AS median_last_response_slots
    FROM action_response_stats GROUP BY action_type
  )
SELECT action_category, action_count, total_votes_cast,
  avg_cc_votes_per_action, avg_cc_participation_pct,
  ROUND((median_first_response_slots / 86400.0)::numeric, 2) AS median_first_response_days,
  ROUND((median_last_response_slots / 86400.0)::numeric, 2) AS median_last_response_days
FROM (SELECT * FROM overall_stats UNION ALL SELECT * FROM by_action_type) combined_stats
ORDER BY CASE WHEN action_category = 'ALL_ACTIONS' THEN 0 ELSE 1 END, action_category
) TO '/tmp/cardano_exports/M3-4-1 CC_response_time_537_609.csv' WITH CSV HEADER;
4.4.2 — Percentage of CC Votes Including Rationale → M3-4-2_CC_rationale_537_609.csv
SQL 
COPY (
WITH
  cc_votes AS (
    SELECT vp.id AS vote_id, ga.type AS action_type, vp.committee_voter AS cc_member_id,
      ch.raw AS cc_member_raw, ch.has_script, vp.vote AS vote_decision,
      b.epoch_no AS vote_epoch, va.url AS anchor_url, va.data_hash AS anchor_hash
    FROM voting_procedure vp
    JOIN gov_action_proposal ga ON vp.gov_action_proposal_id = ga.id
    JOIN committee_hash ch ON vp.committee_voter = ch.id
    JOIN tx ON vp.tx_id = tx.id JOIN block b ON tx.block_id = b.id
    LEFT JOIN voting_anchor va ON vp.voting_anchor_id = va.id
    WHERE vp.committee_voter IS NOT NULL AND b.epoch_no BETWEEN 537 AND 609
  ),
  votes_with_rationale_check AS (
    SELECT *,
      CASE WHEN anchor_url IS NOT NULL AND anchor_url != ''
                AND anchor_hash IS NOT NULL AND anchor_hash != '' THEN true ELSE false END AS has_valid_anchor,
      CASE WHEN vote_decision = 'No' THEN 'No (Unconstitutional)'
           WHEN vote_decision = 'Yes' THEN 'Yes (Constitutional)'
           WHEN vote_decision = 'Abstain' THEN 'Abstain' ELSE 'Other' END AS vote_type_category
    FROM cc_votes
  ),
  combined_stats AS (
    SELECT 'ALL_VOTES'::text AS category, COUNT(*) AS total_votes,
      SUM(CASE WHEN has_valid_anchor THEN 1 ELSE 0 END) AS votes_with_rationale,
      ROUND((SUM(CASE WHEN has_valid_anchor THEN 1 ELSE 0 END)::numeric / NULLIF(COUNT(*),0)::numeric)*100,2) AS pct_with_rationale
    FROM votes_with_rationale_check
    UNION ALL
    SELECT vote_type_category::text, COUNT(*),
      SUM(CASE WHEN has_valid_anchor THEN 1 ELSE 0 END),
      ROUND((SUM(CASE WHEN has_valid_anchor THEN 1 ELSE 0 END)::numeric / NULLIF(COUNT(*),0)::numeric)*100,2)
    FROM votes_with_rationale_check GROUP BY vote_type_category
    UNION ALL
    SELECT ('Action: ' || action_type::text)::text, COUNT(*),
      SUM(CASE WHEN has_valid_anchor THEN 1 ELSE 0 END),
      ROUND((SUM(CASE WHEN has_valid_anchor THEN 1 ELSE 0 END)::numeric / NULLIF(COUNT(*),0)::numeric)*100,2)
    FROM votes_with_rationale_check GROUP BY action_type
  )
SELECT category, total_votes, votes_with_rationale, pct_with_rationale
FROM combined_stats
ORDER BY CASE WHEN category='ALL_VOTES' THEN 1 WHEN category LIKE 'No%' THEN 2
              WHEN category LIKE 'Yes%' THEN 3 WHEN category='Abstain' THEN 4
              WHEN category LIKE 'Action:%' THEN 5 ELSE 6 END, category
) TO '/tmp/cardano_exports/M3-4-2 CC_rationale_537_609.csv' WITH CSV HEADER;
4.4.3 — Constitutional Committee Abstain Rate → M3-4-3_CC_abstain_rate_537_609.csv
SQL 
COPY (
WITH
  cc_votes AS (

    JOIN tx ON vp.tx_id = tx.id JOIN block b ON tx.block_id = b.id

      AND ga.type IN ('TreasuryWithdrawals','ParameterChange','HardForkInitiation','NewConstitution','InfoAction')
  ),

      COUNT(*) AS total_votes_cast,

      SUM(CASE WHEN vote_decision='Yes' THEN 1 ELSE 0 END) AS yes_count,

      SUM(CASE WHEN vote_decision='Abstain' THEN 1 ELSE 0 END),
      SUM(CASE WHEN vote_decision='Yes' THEN 1 ELSE 0 END),
      SUM(CASE WHEN vote_decision='No' THEN 1 ELSE 0 END),

    FROM cc_votes GROUP BY action_type
  )
SELECT category, epoch_no, total_votes_cast, yes_count, no_count, abstain_count, abstain_rate_pct

              WHEN category LIKE 'Action:%' THEN 3 ELSE 4 END,
